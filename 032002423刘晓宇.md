```
def GetDict(str):

    cityDic = { 
    }
    
    localP = re.findall('本土病例\d{1,}例（.*），', str)
    cityP = re.findall('\w{2,}\d{1,}例', localP[0])
    
    for thing in cityP:
        cnt = 0
        for char in thing:
            if(char >= '0' and char <= '9'):
                break
            else:
                cnt += 1

        cityDic[thing[0:cnt]] = thing[cnt:len(thing)-1]

    return cityDic    
```
